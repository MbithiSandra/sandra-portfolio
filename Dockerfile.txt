# Use the official PHP 8.4 image with Apache
FROM php:8.4-apache

# Install system dependencies and PHP extensions Laravel needs
RUN apt-get update && apt-get install -y \
    git \
    curl \
    libpng-dev \
    libonig-dev \
    libxml2-dev \
    libzip-dev \
    zip \
    unzip \
    && docker-php-ext-install pdo pdo_mysql mbstring exif pcntl bcmath gd zip \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Install Composer
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer

# Set working directory
WORKDIR /var/www/html

# Copy all project files into the container
COPY . .

# Install PHP dependencies (no dev packages in production)
RUN composer install --no-dev --optimize-autoloader --no-scripts --no-interaction

# Set correct file permissions so Laravel can write to storage and cache
RUN chown -R www-data:www-data /var/www/html \
    && chmod -R 755 /var/www/html/storage \
    && chmod -R 755 /var/www/html/bootstrap/cache

# Configure Apache to serve from Laravel's public folder
ENV APACHE_DOCUMENT_ROOT /var/www/html/public

RUN sed -ri -e 's!/var/www/html!/var/www/html/public!g' \
    /etc/apache2/sites-available/*.conf \
    && sed -ri -e 's!/var/www/!/var/www/html/public!g' \
    /etc/apache2/conf-available/docker-php.conf \
    /etc/apache2/apache2.conf

# Enable Apache mod_rewrite — Laravel needs this for clean URLs
RUN a2enmod rewrite

# Expose port 80
EXPOSE 80

# Start Apache when container launches
CMD ["apache2-foreground"]
